CREATE DATABASE FoodHubDB1;
GO

USE FoodHubDB1;
GO

/* =========================
   1. CUSTOMER
   ========================= */
CREATE TABLE Customer
(
    CustomerID INT IDENTITY(1,1) PRIMARY KEY,
    CustomerName NVARCHAR(100) NOT NULL,
    NIC NVARCHAR(20) NOT NULL UNIQUE,
    DateOfBirth DATE NOT NULL,
    ContactNumber NVARCHAR(20) NOT NULL,
    LocationNo NVARCHAR(20) NOT NULL,
    Lane NVARCHAR(100) NOT NULL,
    Street NVARCHAR(100) NOT NULL,
    City NVARCHAR(100) NOT NULL,
    CONSTRAINT CK_Customer_DOB CHECK (DateOfBirth < GETDATE())
);
GO

/* =========================
   2. ORDERS
   ========================= */
CREATE TABLE Orders
(
    OrderNo INT IDENTITY(1,1) PRIMARY KEY,
    CustomerID INT NOT NULL,
    OrderDate DATE NOT NULL DEFAULT CAST(GETDATE() AS DATE),
    OrderTime TIME NOT NULL DEFAULT CAST(GETDATE() AS TIME),
    OrderStatus NVARCHAR(20) NOT NULL,
    PaymentMethod NVARCHAR(20) NOT NULL,
    OrderAmount DECIMAL(10,2) NOT NULL DEFAULT 0,
    DispatchTime DATETIME NULL,

    CONSTRAINT FK_Orders_Customer FOREIGN KEY (CustomerID)
        REFERENCES Customer(CustomerID),

    CONSTRAINT CK_Orders_Status CHECK (OrderStatus IN ('Pending', 'Preparing', 'Dispatched', 'Delivered', 'Cancelled')),
    CONSTRAINT CK_Orders_Payment CHECK (PaymentMethod IN ('Cash', 'Card', 'Online')),
    CONSTRAINT CK_Orders_Amount CHECK (OrderAmount >= 0)
);
GO

/* =========================
   3. FOOD ITEM
   ========================= */
CREATE TABLE FoodItem
(
    ItemNumber INT IDENTITY(1,1) PRIMARY KEY,
    ItemName NVARCHAR(100) NOT NULL,
    ItemCategory NVARCHAR(50) NOT NULL,
    Price DECIMAL(10,2) NOT NULL,
    CONSTRAINT UQ_FoodItem_Name UNIQUE (ItemName),
    CONSTRAINT CK_FoodItem_Price CHECK (Price > 0)
);
GO

/* =========================
   4. INGREDIENT
   ========================= */
CREATE TABLE Ingredient
(
    IngredientID INT IDENTITY(1,1) PRIMARY KEY,
    IngredientName NVARCHAR(100) NOT NULL UNIQUE
);
GO

/* =========================
   5. FOOD ITEM INGREDIENT
   ========================= */
CREATE TABLE FoodItemIngredient
(
    ItemNumber INT NOT NULL,
    IngredientID INT NOT NULL,

    PRIMARY KEY (ItemNumber, IngredientID),

    CONSTRAINT FK_FoodItemIngredient_FoodItem FOREIGN KEY (ItemNumber)
        REFERENCES FoodItem(ItemNumber)
        ON DELETE CASCADE,

    CONSTRAINT FK_FoodItemIngredient_Ingredient FOREIGN KEY (IngredientID)
        REFERENCES Ingredient(IngredientID)
        ON DELETE CASCADE
);
GO

/* =========================
   6. RIDER
   ========================= */
CREATE TABLE Rider
(
    EmployeeNumber INT IDENTITY(1,1) PRIMARY KEY,
    FirstName NVARCHAR(50) NOT NULL,
    MiddleName NVARCHAR(50) NULL,
    LastName NVARCHAR(50) NOT NULL,
    NIC NVARCHAR(20) NOT NULL UNIQUE,
    DateOfBirth DATE NOT NULL,
    ContactNumber NVARCHAR(20) NOT NULL,
    LicenseNumber NVARCHAR(50) NOT NULL UNIQUE,
    LocationNo NVARCHAR(20) NOT NULL,
    Lane NVARCHAR(100) NOT NULL,
    Street NVARCHAR(100) NOT NULL,
    City NVARCHAR(100) NOT NULL,
    CONSTRAINT CK_Rider_DOB CHECK (DateOfBirth < GETDATE())
);
GO

/* =========================
   7. DEPENDENT
   ========================= */
CREATE TABLE Dependent
(
    DependentID INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeNumber INT NOT NULL,
    DependentName NVARCHAR(100) NOT NULL,
    RelationshipToRider NVARCHAR(50) NOT NULL,
    DateOfBirth DATE NOT NULL,

    CONSTRAINT FK_Dependent_Rider FOREIGN KEY (EmployeeNumber)
        REFERENCES Rider(EmployeeNumber)
        ON DELETE CASCADE,

    CONSTRAINT CK_Dependent_DOB CHECK (DateOfBirth < GETDATE())
);
GO

/* =========================
   8. MOTOR BIKE
   ========================= */
CREATE TABLE MotorBike
(
    VehicleRegNo NVARCHAR(20) PRIMARY KEY,
    Brand NVARCHAR(50) NOT NULL,
    Model NVARCHAR(50) NOT NULL,
    EngineNumber NVARCHAR(50) NOT NULL UNIQUE,
    VehicleRegisteredDate DATE NOT NULL,
    CONSTRAINT CK_MotorBike_RegDate CHECK (VehicleRegisteredDate <= GETDATE())
);
GO

/* =========================
   9. BIKE COLOUR
   ========================= */
CREATE TABLE BikeColour
(
    VehicleRegNo NVARCHAR(20) NOT NULL,
    ColourName NVARCHAR(50) NOT NULL,

    PRIMARY KEY (VehicleRegNo, ColourName),

    CONSTRAINT FK_BikeColour_MotorBike FOREIGN KEY (VehicleRegNo)
        REFERENCES MotorBike(VehicleRegNo)
        ON DELETE CASCADE
);
GO

/* =========================
   10. BIKE ASSIGNMENT
   ========================= */
CREATE TABLE BikeAssignment
(
    AssignmentID INT IDENTITY(1,1) PRIMARY KEY,
    EmployeeNumber INT NOT NULL,
    VehicleRegNo NVARCHAR(20) NOT NULL,
    AssignmentDate DATE NOT NULL,
    ShiftCode NVARCHAR(20) NOT NULL,
    MeterReadingAssigned DECIMAL(10,2) NOT NULL,
    MeterReadingReturned DECIMAL(10,2) NULL,

    CONSTRAINT FK_BikeAssignment_Rider FOREIGN KEY (EmployeeNumber)
        REFERENCES Rider(EmployeeNumber),

    CONSTRAINT FK_BikeAssignment_MotorBike FOREIGN KEY (VehicleRegNo)
        REFERENCES MotorBike(VehicleRegNo),

    CONSTRAINT CK_BikeAssignment_Shift CHECK (ShiftCode IN ('Morning', 'Evening', 'Night')),
    CONSTRAINT CK_BikeAssignment_MeterAssigned CHECK (MeterReadingAssigned >= 0),
    CONSTRAINT CK_BikeAssignment_MeterReturned CHECK (MeterReadingReturned IS NULL OR MeterReadingReturned >= MeterReadingAssigned)
);
GO

/* =========================
   11. ORDER ITEM
   ========================= */
CREATE TABLE OrderItem
(
    OrderNo INT NOT NULL,
    ItemNumber INT NOT NULL,
    Quantity INT NOT NULL,
    UnitPriceAtOrderTime DECIMAL(10,2) NOT NULL,
    LineAmount AS (Quantity * UnitPriceAtOrderTime) PERSISTED,

    PRIMARY KEY (OrderNo, ItemNumber),

    CONSTRAINT FK_OrderItem_Orders FOREIGN KEY (OrderNo)
        REFERENCES Orders(OrderNo)
        ON DELETE CASCADE,

    CONSTRAINT FK_OrderItem_FoodItem FOREIGN KEY (ItemNumber)
        REFERENCES FoodItem(ItemNumber),

    CONSTRAINT CK_OrderItem_Qty CHECK (Quantity > 0),
    CONSTRAINT CK_OrderItem_UnitPrice CHECK (UnitPriceAtOrderTime > 0)
);
GO


CREATE TRIGGER TRG_UpdateOrderAmount
ON OrderItem
AFTER INSERT, UPDATE, DELETE
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE o
    SET o.OrderAmount = ISNULL(x.TotalAmount, 0)
    FROM Orders o
    OUTER APPLY
    (
        SELECT SUM(oi.LineAmount) AS TotalAmount
        FROM OrderItem oi
        WHERE oi.OrderNo = o.OrderNo
    ) x
    WHERE o.OrderNo IN
    (
        SELECT DISTINCT OrderNo FROM inserted
        UNION
        SELECT DISTINCT OrderNo FROM deleted
    );
END;
GO


CREATE TRIGGER TRG_PreventLessThan3Ingredients
ON FoodItemIngredient
AFTER DELETE
AS
BEGIN
    SET NOCOUNT ON;

    IF EXISTS
    (
        SELECT d.ItemNumber
        FROM deleted d
        CROSS APPLY
        (
            SELECT COUNT(*) AS IngredientCount
            FROM FoodItemIngredient f
            WHERE f.ItemNumber = d.ItemNumber
        ) c
        WHERE c.IngredientCount < 3
    )
    BEGIN
        RAISERROR('Each food item must have at least 3 ingredients.', 16, 1);
        ROLLBACK TRANSACTION;
    END
END;
GO

CREATE VIEW VW_OrderDetails
AS
SELECT 
    o.OrderNo,
    c.CustomerName,
    o.OrderDate,
    o.OrderTime,
    o.OrderStatus,
    o.PaymentMethod,
    o.OrderAmount
FROM Orders o
INNER JOIN Customer c ON o.CustomerID = c.CustomerID;
GO

CREATE VIEW VW_RiderBikeAssignments
AS
SELECT 
    ba.AssignmentID,
    r.EmployeeNumber,
    r.FirstName + ' ' + r.LastName AS RiderName,
    ba.VehicleRegNo,
    ba.AssignmentDate,
    ba.ShiftCode,
    ba.MeterReadingAssigned,
    ba.MeterReadingReturned
FROM BikeAssignment ba
INNER JOIN Rider r ON ba.EmployeeNumber = r.EmployeeNumber;
GO
